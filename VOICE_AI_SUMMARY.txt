USAGE EXAMPLES
==============
# Interactive mode
python3 voice_ai_assistant.py

# Robot voice
python3 voice_ai_assistant.py --preset robot

# Batch mode
python3 voice_ai_assistant.py --batch "Hello" "What time is it?"

# Text-only (no voice)
python3 voice_ai_assistant.py --no-voice

# Run demo
python3 demo_voice_ai.py
